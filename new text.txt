git 기본 명령어

git clone 레포지토리 주소 //내 노트북에 git 폴더가 없을때 폴더 만들어주기

/////////////로그인 할때/////////////////
git config --global user.name"사용자 이름"
git config --global user.email "사용자 이메일"

/////////////내가 git파일안에 어떤 내용을 추가 하고 나서//////////////////
git status//기존의 내용에서 무엇인가 추가했을때 추가된 파일 이름을 보여준다
git add .//내가 폴더안에 뭔가를 추가하고나서 해주는 명령어
git commit // 제목 + 설명  <<<<<<<<<추가해준 파일의 제목과 내가 추가해준 것의 설명을 적어준다.
git push//파일을 올려준다.
git pull//git에 다른사람이 upload 한것을 다운받는다.
